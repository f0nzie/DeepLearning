\begin{algorithm}
  \caption{Logistic Regression on $m$ examples}
  \begin{algorithmic}[1]
    \Initialize{\strut$J \gets 0$, $dw_1 \gets 0$, $dw_2 \gets 0$, $db \gets 0$}
    \For{i = 1 to $m$}
        \State $z^{(i)} = w^T x^{(i)} + b$
        \State $a^{(i)} = \sigma(z^{(i)})$ 
        \State $J$ += $-[y^{(i)} \log a^{(i)} + (1-y^{(i)}) \log(1-a^{(i)})]$   
        \State $dz^{(i)} = a^{(i)} - y^{(i)}$ 
        \State $dw_1$ += $x_1^{(i)} dz^{(i)}$
        \State $dw_2$ += $x_2^{(i)} dz^{(i)}$
        \State $db$ += $dz^{(i)}$
    \EndFor
    \State $J$ /= $m$
    \State $dw_1$ /= $m$; $dw_2$ /=$m$; $db$ /= $m$
  \end{algorithmic}
\end{algorithm}
